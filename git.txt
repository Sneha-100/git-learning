git init: initializing a git repository

git status : list all untracked files

git add-A :adding all untracked files to staging area

git commit -m "commit message": staging area to local repository

git config --global user.name "username"

git config --global user.email "email"

git remote add remote_name remote_path (for adding remote)

git push remote_name branch_name (master/main)

git branch

git checkout
